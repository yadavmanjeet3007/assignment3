def calculate_upper_and_lower(s):
    no_uppercase=0
    no_lowercase=0
    for i in s:
        if i.isupper():
            no_uppercase+=1
        elif i.islower():
            no_lowercase+=1
        else: 
            pass   
    print(s)
    print('No. of uppercase letter: ',no_uppercase)
    print('No. of lowercase letter: ',no_lowercase)
